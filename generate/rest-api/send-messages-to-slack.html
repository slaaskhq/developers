Send messages to Slack

  1. CREATE new BOT message
  2. CREATE new AGENT message

First, learn how to find your secret key.

CREATE new BOT message

Endpoint

POST https://slaask.com/api/messages

Parameters

Key Type Optional? Comments
secret_key String No
conversation_token String No
text String No
from String No Should be bot

Request


curl -X POST "https://slaask.com/api/messages?secret_key=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx&conversation_token=conversation_token&text=text&from=bot"
            

OkHttpClient client = new OkHttpClient();

Request request = new Request.Builder()
  .url("https://slaask.com/api/messages?secret_key=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx&conversation_token=conversation_token&text=text&from=bot")
  .post(null)
  .build();

Response response = client.newCall(request).execute();
            

var data = null;

var xhr = new XMLHttpRequest();
xhr.withCredentials = true;

xhr.addEventListener("readystatechange", function () {
  if (this.readyState === 4) {
    console.log(JSON.parse(this.responseText));
  }
});

xhr.open("POST", "https://slaask.com/api/messages?secret_key=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx&conversation_token=conversation_token&text=text&from=bot");

xhr.send(data);
            

var request = require("request");

var options = { method: 'POST',
  url: 'https://slaask.com/api/messages',
  qs: {
    secret_key: 'xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx',
    conversation_token: 'conversation_token',
    text: 'text'
    from: 'bot'
  };

request(options, function (error, response, body) {
  if (error) throw new Error(error);

  console.log(body);
});
            

setUrl('https://slaask.com/api/messages');
$request->setMethod(HTTP_METH_POST);

$request->setQueryData(array(
  'secret_key' => 'xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx',
  'conversation_token' => 'conversation_token',
  'text' => 'text'
  'from' => 'bot'
));

try {
  $response = $request->send();

  echo $response->getBody();
} catch (HttpException $ex) {
  echo $ex;
}
            

import requests

url = "https://slaask.com/api/messages"

querystring = {"secret_key":"xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx", "conversation_token":"conversation_token", "text": "text, "from": "bot"}

response = requests.request("POST", url, headers={}, params=querystring)

require 'uri'
require 'net/http'

url = URI("https://slaask.com/api/messages?secret_key=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx&conversation_token=conversation_token&text=text&from=bot")

http = Net::HTTP.new(url.host, url.port)
http.use_ssl = true
http.verify_mode = OpenSSL::SSL::VERIFY_NONE

request = Net::HTTP::Post.new(url)

response = http.request(request)
puts response.read_body

Response

Status: 200 OK 


{
  "success": true,
  "result": {
    "content": "Hello world!",
    "from": "Your company Bot",
    "from_image": "https://your-domain.com/bot.png",
    "from_type": "bot",
    "sent_at": "2017-03-13T17:29:52.661Z"
  }
}

Errors

Status: 403 Forbidden 


{
  "success": false,
  "message": "Invalid Widget secret key"
}

Status: 404 Not Found 


{
  "success": false,
  "message": "Unknown Conversation Token"
}

CREATE new AGENT message

Endpoint

POST https://slaask.com/api/messages

Parameters

Key Type Optional? Comments
secret_key String No
conversation_token String No
text String No
from String No Should be agent
agent_slack_id String No Should be the Slack UID of the user.

Request


curl -X POST "https://slaask.com/api/messages?secret_key=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx&conversation_token=conversation_token&text=text&from=agent&agent_slack_id=agent_slack_id"
            

OkHttpClient client = new OkHttpClient();

Request request = new Request.Builder()
  .url("https://slaask.com/api/messages?secret_key=xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx&conversation_token=conversation_token&text=text&from=agent&agent_slack_id=agent_slack_id")
  .post(null)
  .build();

Response response = client.newCall(request).execute();
